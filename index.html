Customizing the default spacing and sizing scale for your project.

tailwind.config.js 
/** @type {import('tailwindcss').config} */
    module.exports = {
    theme:{
        spacing:{
        '1': '8px',
        '2': '12px',
        '3': '16px',
        '4': '24px',
        '5': '32px',
        '6': '48px',
        }
    }
}
